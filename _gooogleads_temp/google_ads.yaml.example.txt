# Google Ads API 配置文件示例
# 将此文件重命名为 google-ads.yaml 并填写您的实际凭据
# 放置在以下位置之一：
# 1. 您的 home 目录 (~/.google-ads.yaml)
# 2. 运行脚本的当前工作目录
# 3. 通过环境变量 GOOGLE_ADS_CONFIGURATION_FILE_PATH 指定的路径

# 开发者令牌 (必填)
# 从 Google Ads API 控制台获取：https://developers.google.com/google-ads/api/docs/concepts/developer-token
developer_token: "YOUR_DEVELOPER_TOKEN"

# OAuth2.0 客户端 ID 和密钥 (必填)
# 从 Google Cloud Console 获取：https://console.cloud.google.com/
client_id: "YOUR_CLIENT_ID.apps.googleusercontent.com"
client_secret: "YOUR_CLIENT_SECRET"

# OAuth2.0 刷新令牌 (必填)
# 使用 Google Ads API 身份验证工具生成：https://developers.google.com/google-ads/api/docs/client-libs/python/authentication
refresh_token: "YOUR_REFRESH_TOKEN"

# 登录客户 ID (必填)
# 您的 Google Ads 经理账号 ID (MCC)，格式为 123-456-7890
login_customer_id: "1234567890"

# 日志级别 (可选)
# 默认为 WARNING，可设置为 DEBUG、INFO、WARNING、ERROR 或 CRITICAL
log_level: "WARNING"

# 日志文件路径 (可选)
# 如果未指定，日志将输出到控制台
# log_file_path: "/path/to/google-ads.log"

# 库版本 (可选)
# 默认为最新版本，建议不要指定除非有特殊需求
# library_version: "v21"

# 使用应用默认凭据 (可选)
# 默认为 false，如果设置为 true，则忽略上述 OAuth2.0 凭据
# use_application_default_credentials: false

# JSON 密钥文件路径 (可选)
# 用于服务账号认证，与上述 OAuth2.0 凭据互斥
# json_key_file_path: "/path/to/service-account-key.json"
