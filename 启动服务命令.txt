# ============================================
# 启动后端服务命令
# ============================================

# 方法一：后台运行（推荐）
nohup uvicorn app.main:app --host 0.0.0.0 --port 8000 > run.log 2>&1 &

# 等待3秒后检查
sleep 3

# 检查服务状态
curl http://127.0.0.1:8000/health

# 查看进程
ps aux | grep uvicorn | grep -v grep

# 查看日志（如果有问题）
tail -n 30 run.log
