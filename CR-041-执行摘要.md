```bash

cd ~/Google-Data-Analysis && git pull origin main

# 2. 停止旧服务
pkill -f 'uvicorn.*app.main' || true && sleep 2

# 3. 启动新服务

nohup uvicorn app.main:app --host 0.0.0.0 --port 8000 > /tmp/backend.log 2>&1 &
sleep 3

# 4. 验证服务
```
