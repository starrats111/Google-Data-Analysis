### 问题3：数据不准确
**症状：** 前端 L7D 数据与 Google Ads 不一致

**排查步骤：**
```bash
# 查看 Google Ads 数据
cd ~/Google-Data-Analysis/backend
python << 'EOF'
from app.database import SessionLocal
from app.models.google_ads_api_data import GoogleAdsApiData
from datetime import date, timedelta

db = SessionLocal()
end_date = date.today() - timedelta(days=1)
begin_date = end_date - timedelta(days=6)

data = db.query(GoogleAdsApiData).filter(
    GoogleAdsApiData.date >= begin_date,
    GoogleAdsApiData.date <= end_date
).all()

print(f'总记录数: {len(data)}')
for d in data[:10]:
    print(f'日期: {d.date}, 费用: {d.cost}, CPC: {d.cpc}, 预算: {d.budget}, 点击: {d.clicks}')
EOF

# 查看平台数据
python << 'EOF'
from app.database import SessionLocal
from app.models.platform_data import PlatformData
from datetime import date, timedelta

db = SessionLocal()
end_date = date.today() - timedelta(days=1)
begin_date = end_date - timedelta(days=6)

data = db.query(PlatformData).filter(
    PlatformData.date >= begin_date,
    PlatformData.date <= end_date
).all()

print(f'总记录数: {len(data)}')
for d in data[:10]:
    print(f'日期: {d.date}, 佣金: {d.commission}, 订单: {d.orders}')
EOF
```
